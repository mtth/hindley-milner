module Types (
  Iden,
  Binding(..), bindingExpr, Expr(..), Lit(..)
) where

import Data.Text (Text)

-- | An identifier.
type Iden = Text

-- | A bound identifier.
data Binding
  = Binding Iden [Iden] Expr
  deriving (Eq, Ord, Show)

bindingExpr :: Binding -> Expr
bindingExpr (Binding _ _ expr) = expr

-- | An expression.
data Expr
  = VarE Iden
  -- ^ A variable (e.g. @abc@).
  | LitE Lit -- 123 or "abc"
  -- ^ A literal value.
  | AppE Expr Expr
  -- ^ An application.
  | LambdaE Iden Expr
  -- ^ A lambda.
  | LetE Binding Expr
  -- ^ A let binding (e.g. @let f x = add x 2 in f 1@).
  deriving (Eq, Ord, Show)

-- | A literal value.
data Lit
  = DoubleL Double
  | StringL Text
  | BoolL Bool
  deriving (Eq, Ord, Show)
