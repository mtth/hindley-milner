{-# LANGUAGE LambdaCase #-}
{-# LANGUAGE OverloadedStrings #-}

-- | This module implements expression evaluation.
module Eval (
  Eval, ValueEnv, Value(..), Builtin, bind, eval
) where

import Infer
import Parse

import Control.Monad.Except (Except, runExcept, throwError)
import Control.Monad.Fix (mfix)
import Control.Monad.Reader (ReaderT, ask, asks, local, runReaderT)
import Data.Map (Map) -- Not strict to allow recursive bindings.
import qualified Data.Map as Map

-- | The expression evaluation monad.
type Eval a = ReaderT ValueEnv (Except TypeError) a

-- | The result of evaluating an expression.
data Value
  = LitV Lit -- ^ Literal value.
  | LambdaV Iden Expr ValueEnv -- ^ Closure.
  | BuiltinV Builtin -- ^ Built-in function.

-- | A binding of identifier to values, in which we can evaluate expressions. Note that we use a
-- lazy map to allow recursive bindings.
type ValueEnv = Map Iden Value

-- | A built-in function.
--
-- Currently, the following functions are available:
--
-- * Unary double operations: @neg@
-- * Binary double operations: @add@, @mul@
-- * Predicates: @eq@, @gt@, @lt@
-- * Conditionals: @if@
data Builtin = Builtin Iden (Expr -> Eval Value)

-- | Adds a binding to the environment, returning the modified environment. Recursive bindings are
-- supported.
addBinding :: Binding -> Eval ValueEnv
addBinding (Binding iden _ expr) = do
  val <- mfix $ \val -> local (Map.insert iden val) $ evalExpr expr
  Map.insert iden val <$> ask

-- | Evaluates an expression.
evalExpr :: Expr -> Eval Value
evalExpr (VarE iden) = asks (Map.lookup iden) >>= \case
  Just val -> pure val
  Nothing -> throwError $ UnboundIden iden
evalExpr (LitE lit) = pure $ LitV lit
evalExpr (AppE expr arg) = evalExpr expr >>= \case
  BuiltinV (Builtin _ f) -> f arg
  LambdaV iden body env -> do
    val <- evalExpr arg
    local (Map.insert iden val . Map.union env) $ evalExpr body
  _ -> throwError $ TypeMismatch undefined undefined -- TODO
evalExpr (LambdaE iden body) = LambdaV iden body <$> ask
evalExpr (LetE binding expr) = do
  env' <- addBinding binding
  local (const env') $ evalExpr expr

bind :: Binding -> ValueEnv -> Either TypeError ValueEnv
bind b env = runExcept $ runReaderT (addBinding b) env

-- | Runs an 'Eval' action.
eval :: ValueEnv -> Expr -> Either TypeError Value
eval env expr = runExcept $ runReaderT (evalExpr expr) env
