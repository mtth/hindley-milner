module Builtins (
  defaultTypeEnv
  -- defaultValueEnv
) where

import Eval
import Infer
import Parse

import Data.Map (Map) -- Not strict to allow recursive bindings.
import qualified Data.Map as Map

defaultTypeEnv :: Map Iden Sig
defaultTypeEnv = Map.fromList
  [ ("add", ForAll [] $ ArrowT doubleType (ArrowT doubleType doubleType))
  , ("neg", ForAll [] $ ArrowT doubleType doubleType)
  , ("gt", ForAll [] $ ArrowT doubleType (ArrowT doubleType boolType))
  , ("lt", ForAll [] $ ArrowT doubleType (ArrowT doubleType boolType))
  , ("if", let v = Var 1; t = VarT v in ForAll [v] $ ArrowT boolType $ ArrowT t $ ArrowT t t)
  ]
